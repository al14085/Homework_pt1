CREATE TABLE new_agents
AS
SELECT *
FROM agents
WHERE WORKING_AREA = "London"
AND customer.CUST_COUNTRY = "UK"
LEFT JOIN customer
ON  customer.WORKING_AREA = new_agents.WORKING_AREA

CREATE TABLE new_agents
AS
SELECT agents.AGENT_NAME, customer.CUST_CODE
FROM agents LEFT JOIN customer
ON  customer.WORKING_AREA = agents.WORKING_AREA 
WHERE agents.WORKING_AREA = "London"
AND customer.CUST_COUNTRY = "UK"
